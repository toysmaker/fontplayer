			const value = table[key]

			}

const getUnicodeRange = (unicode: number) => {
	for (let i = 0; i < unicodeRanges.length; i += 1) {
		const range = unicodeRanges[i]
		if (unicode >= range.begin && unicode < range.end) {
			return i
		}
	}

	return -1
}
